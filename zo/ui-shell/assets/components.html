JavaScript Components

Toast Notifications

// Success notification
window.ZoQoreComponents.toast.success('Operation completed!');

// Error notification
window.ZoQoreComponents.toast.error('An error occurred!');

// Custom toast with options
window.ZoQoreComponents.toast.show({
  title: 'Upload Complete',
  message: 'Your file has been uploaded successfully',
  variant: 'success',
  duration: 5000
});

Modal Dialogs

// Basic modal
window.ZoQoreComponents.modal.show({
  title: 'Modal Title',
  content: '<p>Modal content here</p>',
  actions: [
    { label: 'Cancel', variant: 'secondary', onClick: () => {} },
    { label: 'Save', variant: 'primary', onClick: () => {} }
  ]
});

// Confirmation dialog
const confirmed = await window.ZoQoreComponents.modal.confirm(
  'Are you sure you want to delete this item?',
  { title: 'Confirm Delete', confirmVariant: 'danger' }
);

if (confirmed) {
  // User clicked "Confirm"
}
