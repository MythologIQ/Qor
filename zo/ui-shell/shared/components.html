Modal

// Using the JavaScript API
ZoQoreComponents.modal.show({
  title: 'Modal Title',
  content: '<p>Modal content</p>',
  actions: [
    { label: 'Cancel', variant: 'secondary' },
    { label: 'Confirm', variant: 'primary', onClick: () => console.log('confirmed') }
  ]
});

// Confirm dialog
const result = await ZoQoreComponents.modal.confirm('Are you sure?', {
  title: 'Confirm Action'
});
